AWS ECR — Multi-Platform Image Push

Create a repository, authenticate Docker, then build and push a multi-arch image for both amd64 and arm64 in a single docker buildx command.

# --- Variables ---
ACCOUNT_ID=$(aws sts get-caller-identity --query Account --output text)
REGION="us-east-1"
REPO_NAME="my-app"
IMAGE_TAG="1.0.0"
ECR_URI="${ACCOUNT_ID}.dkr.ecr.${REGION}.amazonaws.com"

# --- 1. Create ECR Repository ---
aws ecr create-repository \
    --repository-name ${REPO_NAME} \
    --region ${REGION} \
    --image-scanning-configuration scanOnPush=true \
    --encryption-configuration encryptionType=AES256

# --- 2. Authenticate Docker to ECR ---
aws ecr get-login-password --region ${REGION} | \
    docker login \
        --username AWS \
        --password-stdin ${ECR_URI}

# --- 3. Create a buildx builder for multi-platform support ---
docker buildx create --name multi-builder --use
docker buildx inspect --bootstrap

# --- 4. Build and push for linux/amd64 + linux/arm64 ---
docker buildx build \
    --platform linux/amd64,linux/arm64 \
    --tag ${ECR_URI}/${REPO_NAME}:${IMAGE_TAG} \
    --tag ${ECR_URI}/${REPO_NAME}:latest \
    --push \
    .
